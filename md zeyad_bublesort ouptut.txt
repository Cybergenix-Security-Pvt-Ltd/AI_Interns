import random

def bubbleSort(arr):
    # iterate from the beginning

    # do not change it
    n=len(arr)
    i=0

    # loop from left to right
    # starting with element = 0
    # set left_flag=1
    # while left_flag==0 then move forward
    # else if left_flag=1 then move left
    while i<n:
        # iterate from the element at index i+1
        for j in range(n-i):
            if arr[j]>arr[j+1] :
                # swap arr[j] and arr[j+1]
                arr[j],arr[j+1]=arr[j+1],arr[j]
        i=i+1

    # do not change it
    return arr
